✓ Cum putem extrage cel mai mic si cel mai mare numar dintr-un sir de numere?

// Prima varianta: functie care contine o instructiune de ciclare care returneaza un obiect ce contine cele 2 valori
function maxMinValue(arr) {
	let max = arr[0];
	let min = arr[0];

	for (let i = 0; i < arr.length; i++) {
		if (arr[i] > max) {
			max = arr[i];
		}

		if (arr[i] < min) {
			min = arr[i];
		}
	}
	return { max, min };
}

// A doua varianta: functie care se foloseste de obiectul Math
function minMaxValue(arr) {
    const min = Math.min.apply(Math, array);
    const max = Math.max.apply(Math, array);
    return { max, min };
}

// A treia varianta: functie care foloseste metoda sort
function minMaxValue(arr) {
    const sirSortat = arr.sort();
    const min = sirSortat[0];
    const max = sirSortat[arr.length - 1];
    return { max, min };
}

	const fileReference = document.createElement("link");

✓ Care sunt diferentele dintre urmatorii operatori: =, == si ===?

  • = este un operator de atribuire a unei valori unei variabile
  • == este un operator de comparatie (nu compara si tipul de date)
  • === este un operator strict de comparatie (compara si tipul de date)
